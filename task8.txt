# -*- coding: utf-8 -*-
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from google.colab import drive
from statsmodels.tsa.seasonal import seasonal_decompose
from sklearn.preprocessing import MinMaxScaler
from keras.preprocessing.sequence import TimeseriesGenerator
from keras.models import Sequential
from keras.layers import Dense, LSTM
from sklearn.metrics import mean_squared_error
from math import sqrt

# 1. Load Data
# Mount drive if needed: drive.mount('/content/drive')
# Assuming the file is available at the path below:
path = '/content/drive/MyDrive/monthlyMilkProduction.csv'
df = pd.read_csv(path, index_col='Date', parse_dates=True)
df.index.freq = 'MS'

# Quick look at the data
print(df.head())
df.plot(figsize=(10, 6))
plt.title("Monthly Milk Production")
plt.show()

# 2. Analysis & Preprocessing
results = seasonal_decompose(df['Production'])
results.plot()
plt.show()

# Split into Train and Test (e.g., last 12 months for testing)
train = df.iloc[:156]
test = df.iloc[156:]

scaler = MinMaxScaler()
scaler.fit(train)

scaled_train = scaler.transform(train)
scaled_test = scaler.transform(test)

# 3. Time Series Generator
n_input = 12 # Look back 12 months to predict the 13th
n_features = 1

generator = TimeseriesGenerator(scaled_train, scaled_train, length=n_input, batch_size=1)

# 4. Define and Train LSTM Model
model = Sequential()
model.add(LSTM(100, activation='relu', input_shape=(n_input, n_features)))
model.add(Dense(1))
model.compile(optimizer='adam', loss='mse')

model.summary()

# Fit model
model.fit(generator, epochs=50)

# Plot Loss
loss_per_epoch = model.history.history['loss']
plt.plot(range(len(loss_per_epoch)), loss_per_epoch)
plt.title("Loss per Epoch")
plt.show()

# 5. Evaluation & Predictions
test_predictions = []

# Get the last n_input points from the training set to start predicting
first_eval_batch = scaled_train[-n_input:]
current_batch = first_eval_batch.reshape((1, n_input, n_features))

for i in range(len(test)):
    # Get prediction (1 time-step ahead)
    current_pred = model.predict(current_batch)[0]
    
    # Store prediction
    test_predictions.append(current_pred)
    
    # Update batch to include prediction and drop first value
    current_batch = np.append(current_batch[:, 1:, :], [[current_pred]], axis=1)

# 6. Inverse Transform and Visualize
true_predictions = scaler.inverse_transform(test_predictions)
test = test.copy() # Avoid SettingWithCopyWarning
test['Predictions'] = true_predictions

print(test.head())
test.plot(figsize=(10, 6))
plt.title("Predictions vs Actuals")
plt.show()

# 7. Calculate Error
rmse = sqrt(mean_squared_error(test['Production'], test['Predictions']))
print(f'RMSE: {rmse}')
