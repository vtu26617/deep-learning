import numpy as np
import os
import cv2
import time
from ultralytics import YOLO
from google.colab.patches import cv2_imshow

# --- Helper: Download sample files if they don't exist ---
if not os.path.exists("dog.jpg"):
    !wget -q https://raw.githubusercontent.com/ultralytics/ultralytics/main/ultralytics/assets/dog.jpg
if not os.path.exists("coco.names"):
    # Creating a quick coco.names file for the standard 80 classes
    classes = ["person", "bicycle", "car", "motorcycle", "airplane", "bus", "train", "truck", "boat", "traffic light", "fire hydrant", "stop sign", "parking meter", "bench", "bird", "cat", "dog", "horse", "sheep", "cow", "elephant", "bear", "zebra", "giraffe", "backpack", "umbrella", "handbag", "tie", "suitcase", "frisbee", "skis", "snowboard", "sports ball", "kite", "baseball bat", "baseball glove", "skateboard", "surfboard", "tennis racket", "bottle", "wine glass", "cup", "fork", "knife", "spoon", "bowl", "banana", "apple", "sandwich", "orange", "broccoli", "carrot", "hot dog", "pizza", "donut", "cake", "chair", "couch", "potted plant", "bed", "dining table", "toilet", "tv", "laptop", "mouse", "remote", "keyboard", "cell phone", "microwave", "oven", "toaster", "sink", "refrigerator", "book", "clock", "vase", "scissors", "teddy bear", "hair drier", "toothbrush"]
    with open("coco.names", "w") as f:
        f.write("\n".join(classes))

# 2. Configuration
CONFIDENCE = 0.5
font_scale = 0.6 # Adjusted for better visibility
thickness = 1

# Loading the YOLOv8 model (n is for 'nano', the fastest version)
model = YOLO("yolov8n.pt")

# Load class labels
labels = open("coco.names").read().strip().split("\n")
colors = np.random.randint(0, 255, size=(len(labels), 3), dtype="uint8")

# 3. Process Image
path_name = "dog.jpg"
image = cv2.imread(path_name)
file_name = os.path.basename(path_name)
filename, ext = os.path.splitext(file_name)

# Measure inference time
start = time.perf_counter()
results = model.predict(image, conf=CONFIDENCE)[0]
time_took = time.perf_counter() - start

print(f"Time took: {time_took:.2f}s")

# 4. Loop over detections and draw
for data in results.boxes.data.tolist():
    # Extract coordinates and info
    xmin, ymin, xmax, ymax, confidence, class_id = data
    xmin, ymin, xmax, ymax, class_id = int(xmin), int(ymin), int(xmax), int(ymax), int(class_id)
    
    color = [int(c) for c in colors[class_id]]
    
    # Draw bounding box
    cv2.rectangle(image, (xmin, ymin), (xmax, ymax), color=color, thickness=2)
    
    # Create label text
    text = f"{labels[class_id]}: {confidence:.2f}"
    (text_width, text_height), baseline = cv2.getTextSize(text, cv2.FONT_HERSHEY_SIMPLEX, fontScale=font_scale, thickness=thickness)
    
    # Background rectangle for text (Opacity effect)
    overlay = image.copy()
    cv2.rectangle(overlay, (xmin, ymin - text_height - 10), (xmin + text_width, ymin), color, -1)
    image = cv2.addWeighted(overlay, 0.6, image, 0.4, 0)
    
    # Put text on image
    cv2.putText(image, text, (xmin, ymin - 5), cv2.FONT_HERSHEY_SIMPLEX, fontScale=font_scale, color=(255, 255, 255), thickness=thickness)

# 5. Display and Save
cv2_imshow(image)
output_path = f"{filename}_yolo8{ext}"
cv2.imwrite(output_path, image)
print(f"Saved to: {output_path}")
